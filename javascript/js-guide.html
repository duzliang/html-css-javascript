Q: typeof null === "object" 为什么是bug？
JavaScript 第一个版本用 32 位存储值，其中 3 位表示类型标签。 object 的类型标签是 000，而 null 在底层表示为全 0（0x00），被误判为 object。 由于修复会破坏大量现有代码，这个 bug 被保留至今。

Q: 以下代码输出什么？为什么？
for (var i = 0; i < 3; i++) {
  setTimeout(() => {
    console.log(i);
  }, 100);
}
// 输出：3, 3, 3

  setTimeout(() => {
    console.log(i);
  }, 100);
}

  • var：函数作用域，循环结束后 i = 3，三个闭包共享同一个 i
  • let：块级作用域，每次迭代创建新的绑定，闭包捕获各自的 i

闭包 (Closure) 核心 必考

什么是闭包

闭包是指函数能够记住并访问它的词法作用域，即使该函数在其词法作用域之外执行。

闭包的形成条件：
  1. 函数嵌套（内部函数）
  2. 内部函数引用外部函数的变量
  3. 内部函数被返回到外部或在别处被调用

经典闭包示例

// 基础闭包
function makeCounter() {
  let count = 0;  // 被闭包保留的变量

  return function() {
    return ++count;  // 访问外部函数的 count
  };
}

const counter = makeCounter();
console.log(counter()); // 1
console.log(counter()); // 2
console.log(counter()); // 3

// count 被保留在内存中，无法直接访问
// 这就是私有变量的实现原理

闭包的实际应用

// 1. 模块模式（封装私有变量）
const myModule = (function() {
  let privateVar = 'I am private';

  function privateMethod() {
    console.log(privateVar);
  }

  return {
    publicMethod: function() {
      privateMethod();
    },
    getPrivateVar: function() {
      return privateVar;
    }
  };
})();

myModule.publicMethod();    // "I am private"
console.log(myModule.getPrivateVar()); // "I am private"

// 2. 函数柯里化
function curry(fn) {
  return function curried(...args) {
    if (args.length >= fn.length) {
      return fn.apply(this, args);
    } else {
      return function(...args2) {
        return curried.apply(this, args.concat(args2));
      };
    }
  };
}

const add = curry((a, b, c) => a + b + c);
console.log(add(1)(2)(3)); // 6
console.log(add(1, 2)(3)); // 6

// 3. 防抖实现（闭包保存 timer）
function debounce(fn, delay) {
  let timer = null;  // 被闭包保存

  return function(...args) {
    clearTimeout(timer);
    timer = setTimeout(() => {
      fn.apply(this, args);
    }, delay);
  };
}

// 4. 缓存（记忆化）
function memoize(fn) {
  const cache = {};  // 被闭包保存

  return function(...args) {
    const key = JSON.stringify(args);
    if (cache[key]) {
      console.log('Cached');
      return cache[key];
    }
    const result = fn.apply(this, args);
    cache[key] = result;
    return result;
  };
}

const fib = memoize(function(n) {
  if (n < 2) return n;
  return fib(n - 1) + fib(n - 2);
});

闭包的陷阱与解决

// 陷阱1：循环中的闭包（旧浏览器）
for (var i = 0; i < 3; i++) {
  setTimeout(function() {
    console.log(i);  // 3, 3, 3
  }, 100);
}

// 解决方案1：IIFE
for (var i = 0; i < 3; i++) {
  (function(j) {
    setTimeout(function() {
      console.log(j);  // 0, 1, 2
    }, 100);
  })(i);
}

// 解决方案2：let（ES6）
for (let i = 0; i < 3; i++) {
  setTimeout(() => {
    console.log(i);  // 0, 1, 2
  }, 100);
}

// 陷阱2：内存泄漏
function leaky() {
  const hugeData = new Array(1000000).fill('x');

  return function() {
    console.log('I hold reference to hugeData');
    // 即使不访问 hugeData，它也被保留在闭包中
  };
}

const leakyFn = leaky();  // hugeData 无法被GC
// 解决：避免在闭包中保留不必要的大对象
Q: 闭包会导致内存泄漏吗？如何避免？

闭包本身不会泄漏，但不当使用会导致变量无法被垃圾回收。

常见场景：

  • DOM 元素引用：闭包中保留 DOM 引用，即使元素已移除
  • 大对象缓存：闭包缓存无限增长
  • 循环引用：旧 IE 浏览器中闭包与 DOM 的循环引用

避免方法：

  • 及时解除引用（fn = null）
  • 使用 WeakMap/WeakSet 存储
  • 避免在闭包中保留不必要的 DOM 引用
  • 限制缓存大小（LRU策略）

this 绑定机制 核心 必考

this 绑定规则优先级

绑定优先级（从高到低）：
  1. new 绑定：this 指向新创建的对象
  2. 显式绑定：call/apply/bind 指定的对象
  3. 隐式绑定：函数作为对象方法调用时的对象
  4. 默认绑定：全局对象（严格模式为 undefined）

四大绑定规则详解

// 1. 默认绑定
function defaultBinding() {
  console.log(this);
}
defaultBinding();  // window（浏览器）/ global（Node）

function strictBinding() {
  'use strict';
  console.log(this);  // undefined
}
strictBinding();

// 2. 隐式绑定
const obj = {
  name: 'Object',
  sayName: function() {
    console.log(this.name);
  }
};
obj.sayName();  // "Object" - this 指向 obj

// 隐式绑定的丢失（经典陷阱）
const say = obj.sayName;
say();  // undefined - this 丢失，变为默认绑定

// 3. 显式绑定
defaultBinding.call(obj);     // "Object"
defaultBinding.apply(obj);    // "Object"
const bound = defaultBinding.bind(obj);
bound();  // "Object"

// call vs apply vs bind
defaultBinding.call(obj, 1, 2, 3);    // 参数列表
defaultBinding.apply(obj, [1, 2, 3]); // 参数数组
const boundLater = defaultBinding.bind(obj, 1, 2);  // 预设参数
boundLater(3);  // 1, 2, 3

// 4. new 绑定
function Person(name) {
  this.name = name;  // this 指向新对象
}
const p = new Person('Alice');
console.log(p.name);  // "Alice"

箭头函数与 this

Q: 手写实现 bind 函数
Function.prototype.myBind = function(context, ...args) {
  const fn = this;

  // 返回的绑定函数
  function bound(...args2) {
    // 如果被 new 调用，this 应该是新对象
    // 否则使用绑定的 context
    const isNew = this instanceof bound;
    return fn.call(isNew ? this : context, ...args, ...args2);
  }

  // 保持原型链
  bound.prototype = Object.create(fn.prototype);

  return bound;
};

// 测试
function greet(greeting, name) {
  console.log(`${greeting}, ${name}! I am ${this.title}`);
}

const boundGreet = greet.myBind({ title: 'Developer' }, 'Hello');
boundGreet('World');  // "Hello, World! I am Developer"

// new 调用
function Person(name) {
  this.name = name;
}
const BoundPerson = Person.myBind({ name: 'bound' });
const p = new BoundPerson('Alice');
console.log(p.name);  // "Alice"（new 优先）

}

Child.prototype = new Parent();

  Parent.call(this);  // 借用构造函数
}

// 3. 组合继承（最常用）
function Child3(name) {
  Parent.call(this);  // 继承属性
}
Child3.prototype = new Parent();  // 继承方法
Child3.prototype.constructor = Child3;  // 修复 constructor

// 4. 寄生组合继承（优化版）
function inherit(Child, Parent) {
  const prototype = Object.create(Parent.prototype);
  prototype.constructor = Child;
  Child.prototype = prototype;
}

function Child4(name) {
  Parent.call(this);
}
inherit(Child4, Parent);

// 5. ES6 Class（语法糖，本质相同）
class ModernChild extends Parent {
  constructor(name) {
    super(name);  // 调用父类构造函数
    this.name = name;
  }

  sayHi() {
    console.log(`Hi, I'm ${this.name}`);
  }
}

// 手写 instanceof
function myInstanceof(obj, Constructor) {
  let proto = Object.getPrototypeOf(obj);
  while (proto) {
    if (proto === Constructor.prototype) return true;
    proto = Object.getPrototypeOf(proto);
  }
  return false;
}

Object.myCreate = function(proto, properties) {

Q: new 操作符具体做了什么？
// 手写 new
function myNew(Constructor, ...args) {
  // 1. 创建空对象，原型指向 Constructor.prototype
  const obj = Object.create(Constructor.prototype);

  // 2. 执行构造函数，this 绑定到新对象
  const result = Constructor.apply(obj, args);

  // 3. 如果构造函数返回对象，则返回该对象；否则返回新对象
  return (result && typeof result === 'object') ? result : obj;
}

// 测试
function Person(name) {
  this.name = name;
}
const p = myNew(Person, 'Alice');
console.log(p.name);  // "Alice"

// 构造函数返回对象的情况
function ReturnObject() {
  this.name = 'not returned';
  return { name: 'returned' };
}
const r = myNew(ReturnObject);
console.log(r.name);  // "returned"

事件循环 (Event Loop) 核心 必考

浏览器事件循环模型

执行顺序：
  1. 执行同步代码（调用栈）
  2. 执行所有微任务（Microtasks）：Promise.then/catch/finally、queueMicrotask、MutationObserver
  3. 执行一个宏任务（Macrotask）：setTimeout/setInterval、setImmediate、I/O、UI渲染
  4. 重复步骤 2-3
// 经典面试题
console.log('1');  // 同步

setTimeout(() => {
  console.log('2');  // 宏任务
}, 0);

Promise.resolve().then(() => {
  console.log('3');  // 微任务
});

console.log('4');  // 同步

// 输出：1, 4, 3, 2

// 复杂版本
console.log('1');

setTimeout(() => {
  console.log('2');
  Promise.resolve().then(() => console.log('3'));
}, 0);

Promise.resolve().then(() => {
  console.log('4');
  setTimeout(() => console.log('5'), 0);
});

console.log('6');

// 输出：1, 6, 4, 2, 3, 5
// 解释：
// - 同步：1, 6
// - 微任务：4（执行中又添加宏任务5）
// - 宏任务：2（执行中又添加微任务3）
// - 微任务：3
// - 宏任务：5

宏任务 vs 微任务

类型 包含 优先级
同步代码 主执行栈 最高
微任务 (Microtask) Promise.then、queueMicrotask、MutationObserver 高
宏任务 (Macrotask) setTimeout/setInterval、setImmediate、I/O、MessageChannel 低
UI 渲染 requestAnimationFrame 每帧一次

Promise 核心

Promise 状态与链式调用

// Promise 状态：pending -> fulfilled/rejected
const p = new Promise((resolve, reject) => {
  setTimeout(() => resolve('success'), 1000);
});

// 链式调用
p
  .then(value => {
    console.log(value);  // "success"
    return value + '!';
  })
  .then(value => {
    console.log(value);  // "success!"
    throw new Error('oops');
  })
  .catch(err => {
    console.log(err.message);  // "oops"
    return 'recovered';
  })
  .then(value => {
    console.log(value);  // "recovered"
  })
  .finally(() => {
    console.log('done');
  });

// then 返回新 Promise
const p2 = Promise.resolve(1);
const p3 = p2.then(v => v + 1);
console.log(p2 === p3);  // false - 新 Promise

Q: 手写 Promise 实现（简化版）
class MyPromise {
  constructor(executor) {
    this.state = 'pending';
    this.value = undefined;
    this.reason = undefined;
    this.onFulfilledCallbacks = [];
    this.onRejectedCallbacks = [];

    const resolve = (value) => {
      if (this.state === 'pending') {
        this.state = 'fulfilled';
        this.value = value;
        this.onFulfilledCallbacks.forEach(fn => fn());
      }
    };

    const reject = (reason) => {
      if (this.state === 'pending') {
        this.state = 'rejected';
        this.reason = reason;
        this.onRejectedCallbacks.forEach(fn => fn());
      }
    };

    try {
      executor(resolve, reject);
    } catch (err) {
      reject(err);
    }
  }

  then(onFulfilled, onRejected) {
    onFulfilled = typeof onFulfilled === 'function' ? onFulfilled : v => v;
    onRejected = typeof onRejected === 'function' ? onRejected : e => { throw e };

    const promise2 = new MyPromise((resolve, reject) => {
      if (this.state === 'fulfilled') {
        setTimeout(() => {
          try {
            const x = onFulfilled(this.value);
            resolvePromise(promise2, x, resolve, reject);
          } catch (err) {
            reject(err);
          }
        }, 0);
      } else if (this.state === 'rejected') {
        setTimeout(() => {
          try {
            const x = onRejected(this.reason);
            resolvePromise(promise2, x, resolve, reject);
          } catch (err) {
            reject(err);
          }
        }, 0);
      } else {
        this.onFulfilledCallbacks.push(() => {
          setTimeout(() => {
            try {
              const x = onFulfilled(this.value);
              resolvePromise(promise2, x, resolve, reject);
            } catch (err) {
              reject(err);
            }
          }, 0);
        });
        this.onRejectedCallbacks.push(() => {
          setTimeout(() => {
            try {
              const x = onRejected(this.reason);
              resolvePromise(promise2, x, resolve, reject);
            } catch (err) {
              reject(err);
            }
          }, 0);
        });
      }
    });

    return promise2;
  }
}

// 处理 then 返回值的辅助函数
function resolvePromise(promise2, x, resolve, reject) {
  if (promise2 === x) {
    reject(new TypeError('Chaining cycle detected'));
    return;
  }
  if (x && (typeof x === 'object' || typeof x === 'function')) {
    let called = false;
    try {
      const then = x.then;
      if (typeof then === 'function') {
        then.call(x, (y) => {
          if (called) return;
          called = true;
          resolvePromise(promise2, y, resolve, reject);
        }, (r) => {
          if (called) return;
          called = true;
          reject(r);
        });
      } else {
        resolve(x);
      }
    } catch (err) {
      if (called) return;
      reject(err);
    }
  } else {
    resolve(x);
  }
}

1. 深拷贝

function deepClone(obj, map = new WeakMap()) {
  // 处理 null 或基本类型
  if (obj === null || typeof obj !== 'object') {
    return obj;
  }

  // 处理 Date
  if (obj instanceof Date) return new Date(obj);

  // 处理 RegExp
  if (obj instanceof RegExp) return new RegExp(obj);

  // 处理 Map
  if (obj instanceof Map) {
    const clone = new Map();
    map.set(obj, clone);
    obj.forEach((value, key) => {
      clone.set(deepClone(key, map), deepClone(value, map));
    });
    return clone;
  }

  // 处理 Set
  if (obj instanceof Set) {
    const clone = new Set();
    map.set(obj, clone);
    obj.forEach(value => {
      clone.add(deepClone(value, map));
    });
    return clone;
  }

  // 处理循环引用
  if (map.has(obj)) {
    return map.get(obj);
  }

  // 创建新对象/数组
  const clone = Array.isArray(obj) ? [] : {};
  map.set(obj, clone);

  // 复制 Symbol 属性
  const symKeys = Object.getOwnPropertySymbols(obj);
  for (const sym of symKeys) {
    clone[sym] = deepClone(obj[sym], map);
  }

  // 复制普通属性
  for (const key in obj) {
    if (obj.hasOwnProperty(key)) {
      clone[key] = deepClone(obj[key], map);
    }
  }

  return clone;
}

2. 节流 Throttle

function throttle(fn, wait, options = {}) {
  let timeout = null;
  let previous = 0;

  return function(...args) {
    const now = Date.now();
    const { leading = true, trailing = true } = options;

    if (!previous && !leading) {
      previous = now;
    }

    const remaining = wait - (now - previous);

    if (remaining <= 0 || remaining > wait) {
      if (timeout) {
        clearTimeout(timeout);
        timeout = null;
      }
      fn.apply(this, args);
      previous = now;
    } else if (!timeout && trailing) {
      timeout = setTimeout(() => {
        previous = leading ? Date.now() : 0;
        timeout = null;
        fn.apply(this, args);
      }, remaining);
    }
  };
}

// leading: 首次立即执行
event.addEventListener('scroll', throttle(handler, 100, { leading: true }));

// trailing: 结束后额外执行一次
event.addEventListener('scroll', throttle(handler, 100, { trailing: true }));

    accumulator = this[0];
    startIndex = 1;
  }

  function flatten(arr, currentDepth) {

// 柯里化
function curry(fn) {
  return function curried(...args) {
    if (args.length >= fn.length) {
      return fn.apply(this, args);
    } else {
      return function(...args2) {
        return curried.apply(this, args.concat(args2));
      };
    }
  };
}

function partial(fn, ...args) {
  return function(...args2) {
    let argIndex = 0;
    const finalArgs = args.map(arg =>
      arg === partial._ ? args2[argIndex++] : arg
    );
    return fn.apply(this, finalArgs.concat(args2.slice(argIndex)));
  };
}

Q: 实现一个发布订阅模式 (EventEmitter)
class EventEmitter {
  constructor() {
    this.events = new Map();
  }

  on(event, listener) {
    if (!this.events.has(event)) {
      this.events.set(event, new Set());
    }
    this.events.get(event).add(listener);
    return () => this.off(event, listener);  // 返回取消订阅函数
  }

  once(event, listener) {
    const wrapper = (...args) => {
      this.off(event, wrapper);
      listener.apply(this, args);
    };
    return this.on(event, wrapper);
  }

  off(event, listener) {
    if (this.events.has(event)) {
      this.events.get(event).delete(listener);
    }
  }

  emit(event, ...args) {
    if (this.events.has(event)) {
      this.events.get(event).forEach(listener => {
        listener.apply(this, args);
      });
    }
  }

  removeAllListeners(event) {
    if (event) {
      this.events.delete(event);
    } else {
      this.events.clear();
    }
  }
}

Q: 实现 LRU 缓存
class LRUCache {
  constructor(capacity) {
    this.capacity = capacity;
    this.cache = new Map();  // 利用 Map 的有序性
  }

  get(key) {
    if (!this.cache.has(key)) {
      return -1;
    }
    // 移到最近使用（删除后重新插入到末尾）
    const value = this.cache.get(key);
    this.cache.delete(key);
    this.cache.set(key, value);
    return value;
  }

  put(key, value) {
    if (this.cache.has(key)) {
      this.cache.delete(key);
    } else if (this.cache.size >= this.capacity) {
      // 删除最久未使用的（第一个）
      const firstKey = this.cache.keys().next().value;
      this.cache.delete(firstKey);
    }
    this.cache.set(key, value);
  }
}
