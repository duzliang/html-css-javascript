TypeScript 知识体系与面试指南 TS 5.4

基础类型系统 核心

TypeScript 独有的类型

类型 说明 示例
any 关闭类型检查，任意类型 let x: any = 4;
unknown 类型安全的 any，需类型收窄后使用 let x: unknown = 4;
never 永不存在的值（用于穷尽检查） function err(): never { throw new Error() }
void 无返回值（undefined 或 null） function log(): void {}
tuple 固定长度、固定类型的数组 let point: [number, number]
enum 枚举类型 enum Color { Red, Green }
literal 字面量类型 let x: 'foo' | 'bar'

any vs unknown 对比

// any：完全放弃类型检查
let notSure: any = 4;
notSure.toFixed();  // OK，但运行时才可能报错
notSure.foo.bar;    // OK，编译不检查

// unknown：类型安全的 any
let maybe: unknown = 4;
// maybe.toFixed();  // Error: 'unknown' 上不存在 'toFixed'

// 必须使用类型收窄
if (typeof maybe === 'number') {
  maybe.toFixed();  // OK，类型已收窄为 number
}

// 类型断言（慎用）
(maybe as number).toFixed();

// unknown 的推荐用法
function parseJSON(json: string): unknown {
  return JSON.parse(json);
}

const data = parseJSON('{"name": "Alice"}');
// 必须类型检查后才能使用
if (data && typeof data === 'object' && 'name' in data) {
  console.log(data.name);
}

Tuple 元组详解

// 基础元组
let point: [number, number] = [10, 20];

// 可选元素（TS 3.0+）
let optional: [number, number?] = [10];

// 剩余元素（TS 4.0+）
let rest: [number, ...string[]] = [1, 'a', 'b'];

// 只读元组
const readonly: readonly [number, string] = [1, 'a'];
// readonly[0] = 2;  // Error

// 具名元组（TS 4.0+）
type Point = [x: number, y: number, z?: number];
const p: Point = [1, 2, 3];

// 解构
const [x, y] = point;

// 越界访问（开启 strict 会报错）
// point[2];  // Error: Tuple type '[number, number]' of length '2' has no element at index '2'

// 元组作为函数参数
type Callback = [string, (data: string) => void];
function handleEvent([event, callback]: Callback) {
  callback(event);
}

Interface 详解 核心

Interface 与 Type Alias 对比

特性 interface type
扩展方式 extends（声明合并） &（交叉类型）
同名声明 自动合并（声明合并） 报错
实现类 class implements 不可
联合/交叉 不支持直接定义 type A = B | C
映射类型 不支持 type A = { [K in T]: V }
计算属性 不支持 支持

抽象类 vs Interface

// 抽象类：可以有实现，有运行时
abstract class Animal {
  abstract name: string;
  abstract makeSound(): void;

  move(): void {  // 可以有实现
    console.log('moving');
  }
}

// 接口：纯结构，无运行时
interface IAnimal {
  name: string;
  makeSound(): void;
}

// 类实现接口
class Dog implements IAnimal {
  name = 'Dog';
  makeSound() {
    console.log('bark');
  }
}

// 类继承抽象类
class Cat extends Animal {
  name = 'Cat';
  makeSound() {
    console.log('meow');
  }
}

// 选择建议：
// - 需要默认实现/运行时检查 -> 抽象类
// - 纯类型约束/需要多重实现 -> 接口
// - 现代 TS 推荐优先使用接口

泛型实战：类型安全的 Event Emitter

type EventMap = {
  [eventName: string]: any[];
};

class TypedEmitter {
  private listeners: { [K in keyof Events]?: Array<(...args: Events[K]) => void> } = {};

  on(event: K, listener: (...args: Events[K]) => void) {
    if (!this.listeners[event]) {
      this.listeners[event] = [];
    }
    this.listeners[event]!.push(listener);
  }

  emit(event: K, ...args: Events[K]) {
    this.listeners[event]?.forEach(listener => listener(...args));
  }
}

// 使用
interface MyEvents {
  data: [string];
  error: [Error];
  complete: [];
}

const emitter = new TypedEmitter();
emitter.on('data', (data) => {
  // data 自动推断为 string
  console.log(data.toUpperCase());
});
emitter.emit('data', 'hello');

类型收窄 Type Narrowing 核心

类型收窄方式

// 1. typeof 收窄
function padLeft(value: string | number) {
  if (typeof value === 'string') {
    return value.padStart(10, '0');  // value 是 string
  }
  return value.toFixed(2);  // value 是 number
}

// 2. 真值收窄
function getLength(str: string | null | undefined) {
  if (str) {
    return str.length;  // str 是 string
  }
  return 0;
}

// 3. instanceof 收窄
function logDate(value: Date | string) {
  if (value instanceof Date) {
    console.log(value.toISOString());
  } else {
    console.log(new Date(value).toISOString());
  }
}

// 4. in 运算符收窄
type Fish = { swim: () => void };
type Bird = { fly: () => void };

function move(animal: Fish | Bird) {
  if ('swim' in animal) {
    animal.swim();  // animal 是 Fish
  } else {
    animal.fly();   // animal 是 Bird
  }
}

// 5. 自定义类型保护（type predicate）
function isFish(animal: Fish | Bird): animal is Fish {
  return (animal as Fish).swim !== undefined;
}

function move2(animal: Fish | Bird) {
  if (isFish(animal)) {
    animal.swim();
  } else {
    animal.fly();
  }
}

// 6. 可辨识联合（Discriminated Union）
type Circle = { kind: 'circle'; radius: number };
type Square = { kind: 'square'; side: number };
type Shape = Circle | Square;

function area(shape: Shape) {
  switch (shape.kind) {
    case 'circle':
      return Math.PI * shape.radius ** 2;  // shape 是 Circle
    case 'square':
      return shape.side ** 2;              // shape 是 Square
    default:
      // exhaustive check
      const _exhaustive: never = shape;
      return _exhaustive;
  }
}

断言函数 (Assertion Functions)

// TS 3.7+ 断言函数
function assertIsString(val: any): asserts val is string {
  if (typeof val !== 'string') {
    throw new Error('Not a string!');
  }
}

function assert(condition: any, msg?: string): asserts condition {
  if (!condition) {
    throw new Error(msg || 'Assertion failed');
  }
}

// 使用
function multiply(x: unknown, y: unknown) {
  assertIsString(x);
  assertIsString(y);
  // x 和 y 在此处都是 string
  return x.concat(y);
}

// 另一种形式
function assertNonNull(val: T): asserts val is NonNullable {
  if (val === undefined || val === null) {
    throw new Error('Value is null or undefined');
  }
}

映射类型 Mapped Types 进阶

基础映射类型

// 基础语法：{ [K in Keys]: ValueType }
type Person = { name: string; age: number };

// 只读版本
type ReadonlyPerson = { readonly [K in keyof Person]: Person[K] };
// { readonly name: string; readonly age: number }

// 可选版本
type PartialPerson = { [K in keyof Person]?: Person[K] };
// { name?: string; age?: number }

// 使用泛型封装
type MyReadonly = { readonly [K in keyof T]: T[K] };
type MyPartial = { [K in keyof T]?: T[K] };
type MyRequired = { [K in keyof T]-?: T[K] };  // -? 移除可选

// 修改属性类型
type Nullable = { [K in keyof T]: T[K] | null };

// key remapping (TS 4.1+)
type Getters = {
  [K in keyof T as `get${Capitalize}`]: () => T[K];
};
type PersonGetters = Getters;
// { getName: () => string; getAge: () => number }

内置映射类型工具

工具类型 作用 实现
Partial<T> 所有属性可选 { [K in keyof T]?: T[K] }
Required<T> 所有属性必选 { [K in keyof T]-?: T[K] }
Readonly<T> 所有属性只读 { readonly [K in keyof T]: T[K] }
Pick<T, K> 选取部分属性 { [P in K]: T[P] }
Omit<T, K> 排除部分属性 Pick<T, Exclude<keyof T, K>>
Record<K, T> 创建字典类型 { [P in K]: T }
Exclude<T, U> 从T中排除U T extends U ? never : T
Extract<T, U> 提取T中的U类型 T extends U ? T : never
NonNullable<T> 排除null/undefined T extends null | undefined ? never : T
ReturnType<T> 提取返回类型 T extends (...) => infer R ? R : any
Parameters<T> 提取参数类型 T extends (...args: infer P) => any ? P : never

实用自定义映射类型

// 深度只读
type DeepReadonly = {
  readonly [K in keyof T]: T[K] extends object
    ? DeepReadonly
    : T[K];
};

// 深度可选
type DeepPartial = {
  [K in keyof T]?: T[K] extends object
    ? DeepPartial
    : T[K];
};

// 扁平化对象（将嵌套转为扁平路径）
type Flatten, Key extends keyof T = keyof T> =
  Key extends string
    ? T[Key] extends Record
      ? { [K in `${Key}.${Flatten}`]: string }
      : { [K in Key]: T[Key] }
    : never;

// 将对象转为 FormData 类型
type FormData = {
  [K in keyof T]: T[K] extends File | Blob
    ? T[K]
    : string;
};

// 为特定属性添加前缀
type WithPrefix = {
  [K in keyof T as `${P}${string & K}`]: T[K];
};

条件类型 Conditional Types 进阶

基础条件类型

// 基本语法：T extends U ? X : Y
type IsString = T extends string ? true : false;

type A = IsString<'hello'>;  // true
type B = IsString<123>;      // false

// 分布式条件类型（Distributive Conditional Types）
type ToArray = T extends any ? T[] : never;

type StrOrNumArray = ToArray;
// string[] | number[] （分发到每个成员）

// 阻止分发

[tpe NoDistribute = [T] extends [any] ? T[] : never;
type ArrayOrNever = NoDistribute;  // (string | number)[]

// 实际应用：过滤类型
type Filter = T extends U ? T : never;
type Strings = Filter;  // string

// 从联合类型中排除
type Exclude = T extends U ? never : T;
type Remaining = Exclude<'a' | 'b' | 'c', 'a' | 'b'>;  // 'c'

infer 关键字

// infer 用于在条件类型中推断类型
type ReturnType = T extends (...args: any[]) => infer R ? R : never;

function greet() {
  return 'hello';
}
type GreetReturn = ReturnType;  // string

// 提取 Promise 的返回类型
type Awaited = T extends Promise ? U : T;
type Result = Awaited>;  // string

// 提取数组元素类型
type ElementType = T extends (infer E)[] ? E : never;
type Num = ElementType;  // number

// 提取函数参数
type Parameters = T extends (...args: infer P) => any ? P : never;

// 提取元组中的类型
type Head = T extends [infer H, ...any[]] ? H : never;
type Tail = T extends [any, ...infer Rest] ? Rest : never;

type First = Head<[1, 2, 3]>;  // 1
type Rest = Tail<[1, 2, 3]>;   // [2, 3]

// 递归 infer 提取嵌套 Promise
type DeepAwaited = T extends Promise
  ? DeepAwaited
  : T;

type DeepResult = DeepAwaited>>;  // string

// 根据类型选择实现
type ApiResponse = T extends { success: true }

  | { success: true; value: T }

4. 类型安全的事件总线

type EventMap = Record;

class EventBus {
  private listeners: {
    [K in keyof Events]?: Array<(...args: Events[K]) => void>;
  } = {};

  on(
    event: K,
    listener: (...args: Events[K]) => void
  ): () => void {
    if (!this.listeners[event]) {
      this.listeners[event] = [];
    }
    this.listeners[event]!.push(listener);

    return () => this.off(event, listener);
  }

  off(
    event: K,
    listener: (...args: Events[K]) => void
  ): void {
    const listeners = this.listeners[event];
    if (listeners) {
      const index = listeners.indexOf(listener);
      if (index > -1) {
        listeners.splice(index, 1);
      }
    }
  }

  emit(event: K, ...args: Events[K]): void {
    this.listeners[event]?.forEach(listener => listener(...args));
  }
}

// 使用
type MyEvents = {
  userLogin: [userId: string, timestamp: number];
  dataUpdate: [payload: { id: string; data: unknown }];
};

const bus = new EventBus();

  // userId: string, timestamp: number
});

        ? [P]
        : []

对比分析 必考

Type vs Interface 选择指南

推荐原则：
  • 优先使用 interface，需要时再使用 type
  • 需要 extends 类或实现 -> interface
  • 需要联合/交叉/映射 -> type
  • 需要条件类型 -> type
  • 简单对象结构 -> interface

extends vs implements

// extends：继承实现
class Animal {
  move() {}
}
class Dog extends Animal {
  bark() {}
}

// implements：实现接口（只有类型约束）
interface CanWalk {
  walk(): void;
}
class Person implements CanWalk {
  walk() {}  // 必须实现
}

// interface extends interface
interface Animal {
  name: string;
}
interface Dog extends Animal {
  breed: string;
}

// class implements 多个 interface
class Service implements CanWalk, CanEat {
  walk() {}
  eat() {}
}

type 和 interface 的合并行为

// interface：声明合并（Declaration Merging）
interface User {
  name: string;
}
interface User {
  age: number;
}
// 结果：User { name: string; age: number; }

// type：重复定义报错
type User = { name: string };
// type User = { age: number };  // Error: Duplicate identifier

// type 通过交叉类型实现类似效果
type User1 = { name: string };
type User2 = { age: number };
type User = User1 & User2;  // { name: string; age: number }
